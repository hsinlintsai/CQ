    <script>

        const display = document.getElementById('display-area');

        

        // 顯示主選單 (自動抓取 data.js 的關卡)

        function showMenu() {

            document.getElementById('main-title').innerText = "理化練習系統";

            let html = '<p>請選擇練習關卡：</p>';

            gameData.forEach((level, index) => {

                html += `<button class="btn" onclick="startLevel(${index})">${level.levelName}</button>`;

            });

            display.innerHTML = html;

        }


        // 開始關卡

        function startLevel(index) {

            const level = gameData[index];

            document.getElementById('main-title').innerText = level.levelName;

            showQuestion(index, 0);

        }


        // 顯示題目

        function showQuestion(lvIdx, qIdx) {

            const q = gameData[lvIdx].questions[qIdx];

            if (!q) {

                display.innerHTML = "<h4>恭喜完成本關！</h4><button class='btn' onclick='showMenu()'>回主選單</button>";

                return;

            }

            

            let html = `<div id="question-text">${q.q}</div>`;

            q.options.forEach((opt, i) => {

                html += `<button class="btn" onclick="checkAns(${lvIdx}, ${qIdx}, ${i})">${opt}</button>`;

            });

            display.innerHTML = html;

        }


        // 檢查答案

        function checkAns(lvIdx, qIdx, userAns) {

            if (userAns === gameData[lvIdx].questions[qIdx].a) {

                alert("答對了！");

                showQuestion(lvIdx, qIdx + 1);

            } else {

                alert("再試一次看看！");

            }

        }


        // 處理密碼 9306696

        function handleClear() {

            const pwd = document.getElementById('pwdInput').value;

            if (pwd === "9306696") {

                alert("數據已清空 (此為模擬動作)");

                document.getElementById('pwdInput').value = "";

            } else {

                alert("密碼不正確");

            }

        }


        // 初始化

        showMenu();

    </script>
